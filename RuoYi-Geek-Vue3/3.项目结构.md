# RuoYi-Geek-Vue3项目结构

```plaintext
├── node_modules                # 第三方依赖库
├── public                      # 静态资源
├── src                         # 源代码文件
│   ├── annotation                # TypeScript
│   │   ├── Api.ts                   # API接口定义
│   │   ├── Schema.ts                # 数据模式定义
│   │   └── SchemaComponent.ts       # Schema组件定义
│   ├── api                       # API请求
│   ├──assets                     # 静态资源
│   ├──components                 # Vue组件
│   ├── directive                 # 自定义指令
│   ├── hook                      # Hook 
│   ├── layout                    # 布局
│   ├── plugins                   # 插件
│   ├── router                    # 路由
│   ├── store                     # Vuex状态管理
│   ├── types                     # 类型定义
│   ├── utils                     # 工具函数
│   └── views                     # 视图
│   ├──App.vue                    # 主入口
│   ├── env.d.ts                  # 环境类型
│   ├── env.d.ts                  # 环境类型
│   ├──main.js                    # 应用启动
│   ├──permission.js              # 权限
│   ├── settings.ts               # 设置
└── vite                        # Vite插件配置

```
